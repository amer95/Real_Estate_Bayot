def extract_transactions_from_pdf(pdf_path):
    """
    Extracts transaction data from a PDF bank statement and returns it as a pandas DataFrame.

    This function was built to processes a PDF file containing tables of transaction data. It extracts
    the date, description, and amount of each transaction, cleans the data by removing 
    empty or invalid rows, and stores the results in a pandas DataFrame.

    Args:
        pdf_path (str): The file path to the PDF bank statement.

    Returns:
        pd.DataFrame: A pandas DataFrame containing the extracted transaction data with 
                      the following columns:
                      - 'Date' (str): The transaction date.
                      - 'Description' (str): The transaction description.
                      - 'Amount' (str): The transaction amount.

    Example:
        >>> df = extract_transactions_from_pdf('Bank-Statement-Template.pdf')
        >>> print(df.head())

    Notes:
        - The PDF should contain tables where each row has transaction information such as 
          date, description, and amount.
        - The function uses pdfplumber (imported as pp) to extract tables from the PDF, so 
          ensure that the pdfplumber library is installed.
        - Newlines and empty rows are removed from the extracted data for clarity.
        - Some Preprocessing maybe needed for the returned data frame.
    """
    
    transactions = []

    with pp.open(pdf_path) as pdf:
        for page in pdf.pages:
            tables = page.extract_tables()  
            for table in tables:
                for row in table:
                    row = [x for x in row if x is not None and '\n' not in x and x != '']
                    
                    if row and len(row) == 1:
                        data = row[0].split(" ", 1)
                        
                        if len(data) > 1:
                            date = data[0]  # Extract the date
                            description_amount = data[1].rsplit(" ", 1) 
                            
                            if len(description_amount) == 2:
                                description, amount = description_amount
                                transactions.append([date, description, amount])  

    df = pd.DataFrame(transactions, columns=['Date', 'Description', 'Amount'])

    return df

def process_loan(df, loan_number):
    """
    Processes loan data from a DataFrame for a specific loan number and returns key statistics.

    This function was built to filters loan disbursements and repayments for a specific loan number from 
    the input DataFrame. It calculates various loan-related metrics such as total disbursed 
    amount, total repaid amount, outstanding balance, most common repayment amount, and the 
    total repayment period in days.

    Args:
        df (pd.DataFrame): The input DataFrame containing loan transaction data. It must 
                           include at least the following columns: 'Description', 'Amount', 'Date'.
        loan_number (int or str): The specific loan number to filter the transactions.

    Returns:
        tuple:
            - loan_df (pd.DataFrame): DataFrame containing all transactions related to the given loan number.
            - loan_repayment (pd.DataFrame): DataFrame containing only repayment transactions for the loan.
            - dict: A dictionary containing the following loan statistics:
                - 'total_disbursed' (float): Total disbursed amount for the loan.
                - 'amount_of_repayment' (float): Most common repayment amount (absolute value).
                - 'number_of_repayments' (int): Number of repayments made for the loan.
                - 'repayment_period_days' (int): The total repayment period in days.
                - 'total_repaid' (float): Total amount repaid for the loan (positive value).
                - 'outstanding_balance' (float): The outstanding balance for the loan (disbursed - repaid).

    Example:
        >>> loan_df, loan_repayment, loan_stats = process_loan(df, 98765)
        >>> print(loan_stats['total_disbursed'])
        >>> print(loan_repayment.head())

    Notes:
        - The function assumes that positive 'Amount' values represent disbursements and negative 'Amount' values 
          represent repayments.
        - The 'Date' column should be in `datetime` format for the function calculations to work correctly.
        - the 'Amount' Column should be numeric (integer or float) for the function calculations to work correctly
        - The outstanding balance is calculated as `total_disbursed + total_repaid`, where `total_repaid` is negative.

    """
    
    loan_df = df[df['Description'].str.contains(str(loan_number), na=False)].copy()

    loan_disbursement = loan_df[loan_df['Amount'] > 0]
    loan_repayment = loan_df[loan_df['Amount'] < 0]
    
    total_disbursed = loan_disbursement['Amount'].sum()
    total_repaid = loan_repayment['Amount'].sum()
    
    outstanding_balance = total_disbursed + total_repaid
    
    period = int(loan_df['Date'].diff().dt.days.sum())
    
    number_of_repayments = loan_repayment.shape[0]
    
    most_common_repayment = abs(loan_repayment['Amount'].value_counts().index[0])
    
    return loan_df, loan_repayment, {
        'total_disbursed': total_disbursed,
        'amount_of_repayment': most_common_repayment,
        'number_of_repayments': number_of_repayments,
        'repayment_period_days': period,
        'total_repaid': abs(total_repaid),
        'outstanding_balance': outstanding_balance,
    }
